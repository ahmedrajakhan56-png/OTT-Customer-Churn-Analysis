```python
"""
OTT CUSTOMER CHURN ANALYSIS - PYTHON ETL
Business Goal: Clean 5,000+ customer records, create features for churn prediction
Tools in Service of Business: Python helps us understand customer behavior
"""

import pandas as pd
import numpy as np
import matplotlib.pyplot as plt
import seaborn as sns
from datetime import datetime, timedelta

# ============================================
# 1. LOAD DATA - 5,000+ customers
# ============================================
print("🚀 Loading customer data...")
df = pd.read_csv('customer_churn_data.csv')
print(f"📊 Dataset Shape: {df.shape}")  # (5,043, 35)
print(f"📅 Data Period: {df['signup_date'].min()} to {df['signup_date'].max()}")

# ============================================
# 2. DATA CLEANING - Because dirty data = wrong decisions
# ============================================
print("\n🧹 Cleaning data for accurate business insights...")

# Check for missing values
print(f"❓ Missing values before cleaning:")
print(df.isnull().sum())

# Handle missing values
# For numeric columns: fill with median
numeric_cols = ['age', 'watch_hours_weekly', 'num_devices', 'payment_amount']
for col in numeric_cols:
    df[col].fillna(df[col].median(), inplace=True)

# For categorical: fill with mode
categorical_cols = ['gender', 'subscription_plan', 'payment_method', 'region']
for col in categorical_cols:
    df[col].fillna(df[col].mode()[0], inplace=True)

# Remove duplicates
initial_count = len(df)
df.drop_duplicates(subset=['customer_id'], inplace=True)
dupes_removed = initial_count - len(df)
print(f"✅ Removed {dupes_removed} duplicate customers")

# ============================================
# 3. FEATURE ENGINEERING - Creating churn predictors
# ============================================
print("\n⚙️ Creating customer behavior features...")

# Calculate customer tenure (days since signup)
df['signup_date'] = pd.to_datetime(df['signup_date'])
df['last_login'] = pd.to_datetime(df['last_login'])
df['tenure_days'] = (datetime.now() - df['signup_date']).dt.days
df['days_since_last_login'] = (datetime.now() - df['last_login']).dt.days

# Create engagement score (0-100)
df['engagement_score'] = (
    df['watch_hours_weekly'] / 40 * 40 +  # Max 40 hours = 40 points
    df['num_devices'] / 5 * 30 +           # Max 5 devices = 30 points
    (30 - df['days_since_last_login']) * 1  # Recent login = 30 points
).clip(0, 100).round()

# Create churn flag (1 if churned, 0 if active)
df['churn'] = df['churn_status'].map({'Yes': 1, 'No': 0})

# Create customer segments
df['customer_segment'] = pd.cut(
    df['engagement_score'], 
    bins=[0, 30, 60, 80, 100],
    labels=['Lost Cause', 'Low Engagement', 'At Risk', 'High Value']
)

# Create tenure groups
df['tenure_group'] = pd.cut(
    df['tenure_days'],
    bins=[0, 30, 90, 180, 365, 9999],
    labels=['<1 month', '1-3 months', '3-6 months', '6-12 months', '>1 year']
)

# ============================================
# 4. BUSINESS INSIGHTS - Quick Python analysis
# ============================================
print("\n💡 Quick Business Insights from Python:")

# Overall churn rate
churn_rate = df['churn'].mean() * 100
print(f"📉 Overall Churn Rate: {churn_rate:.1f}%")

# Churn by subscription plan
churn_by_plan = df.groupby('subscription_plan')['churn'].mean() * 100
print("\n📊 Churn by Subscription Plan:")
print(churn_by_plan.sort_values(ascending=False))

# Churn by region
churn_by_region = df.groupby('region')['churn'].mean() * 100
print("\n🌍 Churn by Region:")
print(churn_by_region.sort_values(ascending=False))

# Top 3 churn drivers correlation
correlation = df[['churn', 'days_since_last_login', 'watch_hours_weekly', 
                  'payment_amount', 'num_devices', 'support_tickets']].corr()['churn'].sort_values(ascending=False)
print("\n🔑 Top Churn Drivers (Correlation):")
print(correlation)

# ============================================
# 5. VISUALIZE - Quick insights before Power BI
# ============================================
fig, axes = plt.subplots(2, 2, figsize=(15, 10))

# Churn by plan
churn_by_plan.plot(kind='bar', ax=axes[0,0], color=['#E50914' if x > 30 else '#34A853' for x in churn_by_plan])
axes[0,0].set_title('Churn Rate by Subscription Plan', fontsize=14)
axes[0,0].set_ylabel('Churn Rate (%)')
axes[0,0].tick_params(axis='x', rotation=45)

# Days since last login vs churn
axes[0,1].hist([df[df['churn']==0]['days_since_last_login'], 
                df[df['churn']==1]['days_since_last_login']], 
                bins=30, label=['Active', 'Churned'], color=['#34A853', '#E50914'], alpha=0.7)
axes[0,1].set_title('Days Since Last Login: Active vs Churned', fontsize=14)
axes[0,1].set_xlabel('Days Since Last Login')
axes[0,1].legend()

# Watch hours distribution
axes[1,0].boxplot([df[df['churn']==0]['watch_hours_weekly'], 
                   df[df['churn']==1]['watch_hours_weekly']], 
                   labels=['Active', 'Churned'])
axes[1,0].set_title('Weekly Watch Hours: Active vs Churned', fontsize=14)
axes[1,0].set_ylabel('Hours')

# Customer segments distribution
segment_counts = df['customer_segment'].value_counts()
colors = ['#E50914' if x == 'Lost Cause' else '#FBBC05' if x == 'Low Engagement' 
          else '#4285F4' if x == 'At Risk' else '#34A853' for x in segment_counts.index]
axes[1,1].pie(segment_counts.values, labels=segment_counts.index, autopct='%1.1f%%', colors=colors)
axes[1,1].set_title('Customer Segment Distribution', fontsize=14)

plt.tight_layout()
plt.savefig('python_churn_insights.png', dpi=100)
print("\n✅ Quick insights visualization saved!")

# ============================================
# 6. EXPORT CLEAN DATA - Ready for SQL, AWS, Power BI
# ============================================
df.to_csv('customer_churn_clean.csv', index=False)
print("\n✅ Clean data exported for SQL/AWS/Power BI")

# Business Summary
print("\n" + "="*50)
print("🎯 BUSINESS SUMMARY (from Python)")
print("="*50)
print(f"Total Customers Analyzed: {len(df):,}")
print(f"Churned Customers: {df['churn'].sum():,}")
print(f"Active Customers: {len(df) - df['churn'].sum():,}")
print(f"Overall Churn Rate: {churn_rate:.1f}%")
print(f"Highest Risk Segment: {churn_by_plan.idxmax()} ({churn_by_plan.max():.1f}% churn)")
print(f"Annual Revenue at Risk: ₹{df[df['churn']==1]['annual_revenue'].sum():,.2f}")
print("="*50)
```

```sql
/*
OTT CUSTOMER CHURN ANALYSIS - SQL SERVER QUERIES
Business Goal: Answer specific business questions about customer behavior
Focus: Churn patterns, customer segments, revenue impact
*/

-- ============================================
-- 1. DATABASE SETUP - Customer Data Warehouse
-- ============================================
CREATE DATABASE OTTChurnAnalytics;
GO

USE OTTChurnAnalytics;
GO

-- Main customers table
CREATE TABLE Customers (
    CustomerID INT PRIMARY KEY,
    Age INT,
    Gender VARCHAR(10),
    Region VARCHAR(50),
    SubscriptionPlan VARCHAR(20),
    PaymentMethod VARCHAR(20),
    MonthlyPayment DECIMAL(10,2),
    AnnualRevenue DECIMAL(10,2),
    SignupDate DATE,
    LastLogin DATE,
    WatchHoursWeekly DECIMAL(5,2),
    NumDevices INT,
    SupportTickets INT,
    Churn BIT,
    EngagementScore INT,
    CustomerSegment VARCHAR(20),
    TenureDays INT,
    DaysSinceLastLogin INT,
    CreatedDate DATETIME DEFAULT GETDATE()
);

-- Load data (from Python cleaned file)
BULK INSERT Customers
FROM 'C:\data\customer_churn_clean.csv'
WITH (
    FIRSTROW = 2,
    FIELDTERMINATOR = ',',
    ROWTERMINATOR = '\n',
    TABLOCK
);

-- ============================================
-- 2. BUSINESS QUESTION: What's the overall churn picture?
-- ============================================
SELECT 
    COUNT(*) AS TotalCustomers,
    SUM(Churn) AS ChurnedCustomers,
    CAST(100.0 * SUM(Churn) / COUNT(*) AS DECIMAL(5,1)) AS ChurnRate,
    SUM(CASE WHEN Churn = 1 THEN AnnualRevenue ELSE 0 END) AS RevenueAtRisk,
    AVG(CASE WHEN Churn = 1 THEN TenureDays ELSE NULL END) AS AvgTenureChurned,
    AVG(CASE WHEN Churn = 0 THEN TenureDays ELSE NULL END) AS AvgTenureActive
FROM Customers;

-- ============================================
-- 3. BUSINESS QUESTION: Which customer segments churn most?
-- ============================================
SELECT 
    CustomerSegment,
    COUNT(*) AS CustomerCount,
    SUM(Churn) AS ChurnedCustomers,
    CAST(100.0 * SUM(Churn) / COUNT(*) AS DECIMAL(5,1)) AS ChurnRate,
    SUM(AnnualRevenue) AS TotalRevenue,
    SUM(CASE WHEN Churn = 1 THEN AnnualRevenue ELSE 0 END) AS RevenueAtRisk
FROM Customers
GROUP BY CustomerSegment
ORDER BY ChurnRate DESC;

-- ============================================
-- 4. BUSINESS QUESTION: Impact of subscription plans
-- ============================================
SELECT 
    SubscriptionPlan,
    COUNT(*) AS CustomerCount,
    SUM(Churn) AS ChurnedCustomers,
    CAST(100.0 * SUM(Churn) / COUNT(*) AS DECIMAL(5,1)) AS ChurnRate,
    AVG(WatchHoursWeekly) AS AvgWatchHours,
    AVG(DaysSinceLastLogin) AS AvgDaysSinceLogin,
    -- Risk level
    CASE 
        WHEN AVG(Churn) * 100 > 50 THEN '🔴 CRITICAL'
        WHEN AVG(Churn) * 100 BETWEEN 30 AND 50 THEN '🟡 HIGH'
        WHEN AVG(Churn) * 100 BETWEEN 15 AND 30 THEN '🟢 MEDIUM'
        ELSE '✅ LOW'
    END AS RiskLevel
FROM Customers
GROUP BY SubscriptionPlan
ORDER BY ChurnRate DESC;

-- ============================================
-- 5. BUSINESS QUESTION: Regional churn patterns
-- ============================================
SELECT 
    Region,
    COUNT(*) AS CustomerCount,
    SUM(Churn) AS ChurnedCustomers,
    CAST(100.0 * SUM(Churn) / COUNT(*) AS DECIMAL(5,1)) AS ChurnRate,
    AVG(EngagementScore) AS AvgEngagement,
    SUM(AnnualRevenue) AS TotalRevenue,
    -- Regional ranking
    RANK() OVER (ORDER BY SUM(Churn) * 1.0 / COUNT(*) DESC) AS ChurnRank
FROM Customers
GROUP BY Region
ORDER BY ChurnRate DESC;

-- ============================================
-- 6. BUSINESS QUESTION: Tenure-based churn analysis
-- ============================================
WITH TenureSegments AS (
    SELECT 
        CASE 
            WHEN TenureDays < 30 THEN '1. <1 month'
            WHEN TenureDays BETWEEN 30 AND 90 THEN '2. 1-3 months'
            WHEN TenureDays BETWEEN 91 AND 180 THEN '3. 3-6 months'
            WHEN TenureDays BETWEEN 181 AND 365 THEN '4. 6-12 months'
            ELSE '5. >1 year'
        END AS TenureSegment,
        Churn,
        AnnualRevenue,
        CustomerID
    FROM Customers
)
SELECT 
    TenureSegment,
    COUNT(*) AS CustomerCount,
    SUM(Churn) AS ChurnedCustomers,
    CAST(100.0 * SUM(Churn) / COUNT(*) AS DECIMAL(5,1)) AS ChurnRate,
    SUM(CASE WHEN Churn = 1 THEN AnnualRevenue ELSE 0 END) AS RevenueAtRisk
FROM TenureSegments
GROUP BY TenureSegment
ORDER BY TenureSegment;

-- ============================================
-- 7. STORED PROCEDURE: High-Risk Customers Report
-- ============================================
CREATE PROCEDURE sp_GetHighRiskCustomers
    @RiskThreshold INT = 70  -- Engagement score below this = high risk
AS
BEGIN
    SELECT 
        CustomerID,
        Age,
        Region,
        SubscriptionPlan,
        WatchHoursWeekly,
        DaysSinceLastLogin,
        EngagementScore,
        AnnualRevenue,
        -- Recommendation based on patterns
        CASE 
            WHEN DaysSinceLastLogin > 30 THEN 'Send re-engagement email'
            WHEN WatchHoursWeekly < 5 THEN 'Recommend popular content'
            WHEN SupportTickets > 3 THEN 'Schedule check-in call'
            ELSE 'Monitor closely'
        END AS RecommendedAction
    FROM Customers
    WHERE Churn = 0  -- Still active but at risk
        AND EngagementScore < @RiskThreshold
    ORDER BY EngagementScore;
END;
GO

-- Execute: EXEC sp_GetHighRiskCustomers 70;

-- ============================================
-- 8. BUSINESS QUESTION: Payment method impact
-- ============================================
SELECT 
    PaymentMethod,
    COUNT(*) AS CustomerCount,
    SUM(Churn) AS ChurnedCustomers,
    CAST(100.0 * SUM(Churn) / COUNT(*) AS DECIMAL(5,1)) AS ChurnRate,
    AVG(MonthlyPayment) AS AvgPayment,
    -- Churn risk category
    CASE 
        WHEN AVG(Churn) * 100 > 40 THEN 'High Risk'
        WHEN AVG(Churn) * 100 BETWEEN 25 AND 40 THEN 'Medium Risk'
        ELSE 'Low Risk'
    END AS RiskCategory
FROM Customers
GROUP BY PaymentMethod
ORDER BY ChurnRate DESC;

-- ============================================
-- 9. BUSINESS VIEW - For Power BI Dashboard
-- ============================================
CREATE VIEW vw_ChurnDashboard AS
SELECT 
    CustomerID,
    Age,
    CASE 
        WHEN Age < 25 THEN '18-24'
        WHEN Age BETWEEN 25 AND 34 THEN '25-34'
        WHEN Age BETWEEN 35 AND 44 THEN '35-44'
        ELSE '45+'
    END AS AgeGroup,
    Gender,
    Region,
    SubscriptionPlan,
    PaymentMethod,
    MonthlyPayment,
    AnnualRevenue,
    WatchHoursWeekly,
    NumDevices,
    SupportTickets,
    DaysSinceLastLogin,
    EngagementScore,
    CustomerSegment,
    TenureDays,
    Churn,
    -- Churn flag for calculations
    CAST(Churn AS INT) AS IsChurned,
    -- Revenue at risk
    CASE WHEN Churn = 1 THEN AnnualRevenue ELSE 0 END AS RevenueLost
FROM Customers;
GO

-- ============================================
-- 10. BUSINESS SUMMARY REPORT
-- ============================================
SELECT '👥 Total Customers' AS Metric, FORMAT(COUNT(*), 'N0') AS Value FROM Customers
UNION ALL
SELECT '📉 Churn Rate', FORMAT(100.0 * SUM(Churn) / COUNT(*), 'N1') + '%' FROM Customers
UNION ALL
SELECT '💰 Revenue at Risk', '₹' + FORMAT(SUM(CASE WHEN Churn = 1 THEN AnnualRevenue ELSE 0 END), 'N0') FROM Customers
UNION ALL
SELECT '🎯 Highest Risk Segment', 
    (SELECT TOP 1 CustomerSegment FROM Customers GROUP BY CustomerSegment ORDER BY AVG(Churn * 1.0) DESC)
UNION ALL
SELECT '📊 Avg Engagement Score', FORMAT(AVG(EngagementScore), 'N0') FROM Customers;
```

```python
"""
OTT CUSTOMER CHURN ANALYSIS - AWS IMPLEMENTATION
Business Goal: Enable team-wide access to customer data, run ad-hoc queries
Focus: Data lake on S3, serverless queries with Athena
"""

import boto3
import pandas as pd
import json
from io import StringIO
import time

# ============================================
# 1. AWS S3 SETUP - Central Data Lake
# ============================================
print("🚀 Setting up AWS S3 data lake for customer data...")

# Configure AWS (using credentials from environment)
s3_client = boto3.client(
    's3',
    region_name='ap-south-1'  # Mumbai region
)

# Create bucket for customer data
bucket_name = 'ott-churn-analytics-6663'
try:
    s3_client.create_bucket(
        Bucket=bucket_name,
        CreateBucketConfiguration={'LocationConstraint': 'ap-south-1'}
    )
    print(f"✅ Created bucket: {bucket_name}")
except:
    print(f"ℹ️ Bucket {bucket_name} already exists")

# ============================================
# 2. ORGANIZE DATA IN S3 (Like folders for business units)
# ============================================
print("\n📁 Organizing customer data in S3...")

# Structure: s3://bucket/category/year/month/file.csv
folders = [
    'raw/customers/',
    'raw/transactions/',
    'raw/viewing_history/',
    'processed/daily/',
    'processed/monthly/',
    'reports/churn_analysis/',
    'reports/executive/',
    'ml_features/',
    'athena_results/'
]

for folder in folders:
    s3_client.put_object(
        Bucket=bucket_name,
        Key=folder
    )
print("✅ Folder structure created")

# ============================================
# 3. UPLOAD DATA TO S3
# ============================================
print("\n📤 Uploading customer data to S3...")

# Read cleaned data from Python
df = pd.read_csv('customer_churn_clean.csv')

# Convert to CSV in memory
csv_buffer = StringIO()
df.to_csv(csv_buffer, index=False)

# Upload raw data
s3_client.put_object(
    Bucket=bucket_name,
    Key='raw/customers/customer_data_2026.csv',
    Body=csv_buffer.getvalue()
)

# Also upload partitioned by region (for faster queries)
for region in df['region'].unique():
    region_df = df[df['region'] == region]
    region_buffer = StringIO()
    region_df.to_csv(region_buffer, index=False)
    
    s3_client.put_object(
        Bucket=bucket_name,
        Key=f'processed/daily/region={region}/customers.csv',
        Body=region_buffer.getvalue()
    )
print(f"✅ Uploaded {len(df):,} customer records to S3")

# ============================================
# 4. AWS ATHENA SETUP - Serverless SQL Queries
# ============================================
print("\n🔍 Setting up AWS Athena for serverless queries...")

athena_client = boto3.client('athena', region_name='ap-south-1')

# Create database in Athena
create_db_query = "CREATE DATABASE IF NOT EXISTS ott_churn_analytics;"

response = athena_client.start_query_execution(
    QueryString=create_db_query,
    QueryExecutionContext={'Database': 'ott_churn_analytics'},
    ResultConfiguration={'OutputLocation': f's3://{bucket_name}/athena_results/'}
)

print("✅ Athena database created")

# ============================================
# 5. CREATE EXTERNAL TABLE IN ATHENA
# ============================================
print("\n📊 Creating external table for customer data...")

create_table_query = f"""
CREATE EXTERNAL TABLE IF NOT EXISTS customers (
    customer_id INT,
    age INT,
    gender STRING,
    region STRING,
    subscription_plan STRING,
    payment_method STRING,
    monthly_payment DOUBLE,
    annual_revenue DOUBLE,
    signup_date DATE,
    last_login DATE,
    watch_hours_weekly DOUBLE,
    num_devices INT,
    support_tickets INT,
    churn INT,
    engagement_score INT,
    customer_segment STRING,
    tenure_days INT,
    days_since_last_login INT
)
ROW FORMAT SERDE 'org.apache.hadoop.hive.serde2.lazy.LazySimpleSerDe'
WITH SERDEPROPERTIES (
    'serialization.format' = ',',
    'field.delim' = ','
)
LOCATION 's3://{bucket_name}/raw/customers/'
TBLPROPERTIES ('has_encrypted_data'='false');
"""

response = athena_client.start_query_execution(
    QueryString=create_table_query,
    QueryExecutionContext={'Database': 'ott_churn_analytics'},
    ResultConfiguration={'OutputLocation': f's3://{bucket_name}/athena_results/'}
)

# Wait for query to complete
execution_id = response['QueryExecutionId']
time.sleep(3)

print("✅ External table created - now you can query directly from S3!")

# ============================================
# 6. BUSINESS QUERIES IN ATHENA
# ============================================
print("\n💡 Running business queries in Athena...")

# Business Query 1: Churn rate by region
query1 = """
SELECT 
    region,
    COUNT(*) as total_customers,
    SUM(churn) as churned_customers,
    ROUND(SUM(churn) * 100.0 / COUNT(*), 2) as churn_rate
FROM customers
GROUP BY region
ORDER BY churn_rate DESC;
"""

response = athena_client.start_query_execution(
    QueryString=query1,
    QueryExecutionContext={'Database': 'ott_churn_analytics'},
    ResultConfiguration={'OutputLocation': f's3://{bucket_name}/athena_results/'}
)
print("✅ Query 1 submitted: Churn by region")

# Business Query 2: High-risk active customers
query2 = """
SELECT 
    customer_id,
    region,
    subscription_plan,
    watch_hours_weekly,
    days_since_last_login,
    engagement_score
FROM customers
WHERE churn = 0 
    AND engagement_score < 50
ORDER BY engagement_score
LIMIT 100;
"""

response = athena_client.start_query_execution(
    QueryString=query2,
    QueryExecutionContext={'Database': 'ott_churn_analytics'},
    ResultConfiguration={'OutputLocation': f's3://{bucket_name}/athena_results/'}
)
print("✅ Query 2 submitted: High-risk active customers")

# Business Query 3: Revenue impact
query3 = """
SELECT 
    customer_segment,
    COUNT(*) as customer_count,
    SUM(churn) as churned,
    ROUND(SUM(churn) * 100.0 / COUNT(*), 2) as churn_rate,
    ROUND(SUM(CASE WHEN churn = 1 THEN annual_revenue ELSE 0 END), 2) as revenue_at_risk
FROM customers
GROUP BY customer_segment
ORDER BY revenue_at_risk DESC;
"""

response = athena_client.start_query_execution(
    QueryString=query3,
    QueryExecutionContext={'Database': 'ott_churn_analytics'},
    ResultConfiguration={'OutputLocation': f's3://{bucket_name}/athena_results/'}
)
print("✅ Query 3 submitted: Revenue impact by segment")

print("\n✅ AWS Setup Complete!")
print(f"📊 Your customer data is now in S3: s3://{bucket_name}/")
print("🔍 Query directly with Athena - no servers to manage!")
print("📈 Power BI can connect directly to Athena for live dashboards")
```
